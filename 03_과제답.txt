use world;
국가코드(countrycode)가 'KOR'인 도시(city)를 찾아 인구수(population)를 역순으로 표시하세요
SELECT  * 
FROM city
WHERE countrycode='KOR'
ORDER BY population DESC;

국가코드와 인구수를 출력, 국가코드별 오름차순, 동일국가안에서 인구수의 역순으로 출력하세요
SELECT *
FROM city
ORDER BY countrycode ASC, population DESC;

국가코드(countrycode)가 'KOR'인 도시수를 출력하시오
SELECT COUNT(*)
FROM city
WHERE countrycode='KOR';

국가코드(countrycode)가 'KOR', 'CHN', 'JPN'인 도시들을 출력하시오
SELECT *
FROM city
WHERE countrycode IN ('KOR', 'CHN', 'JPN');

국가코드(countrycode)가 'KOR'이고 인구가 100만이상인 도시를 출력하시오
SELECT *
FROM city
WHERE countrycode ='KOR' AND population >= 1000000;

국가코드(countrycode)가 'KOR'이고 인구수가 많은 순서로 상위 10개만 출력하시오
SELECT *
FROM city
WHERE countrycode ='KOR'
ORDER BY population  DESC
LIMIT 10;

국가코드(countrycode)가 'KOR'이고 인구수가 100만이상 500만 이하인 도시를 출력하시오
SELECT *
FROM city
WHERE countrycode ='KOR'
AND population BETWEEN   1000000 AND 5000000;
